static Future<void> addUser(String userId, String username, String email) async {
  try {
    await FirebaseFirestore.instance.collection('users').doc(userId).set({
      'username': username,
      'email': email,
    });
  } catch (e) {
    print('Error: adding user: $e\n');
  }
}
